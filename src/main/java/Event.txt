class ToDO extends Event{
    String description;
    boolean isDone;
    int index;
    String time;

    public ToDO(String description, boolean isDone,int index,String type,String time){
        super(description,isDOne,index);
        this.type = "Event";
        this.time = time;
    }
    public String toString(){
        return(this.index + "[" + "E" + "][" + this.getStatusIcon() + "] " + this.description);
    }
}
